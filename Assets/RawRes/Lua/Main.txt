--绑定事件
function Main_AddEvent()

	Main_AddBtn(btnBattle);
end

function Main_AddBtn(btn)
	UIEventListener.AddEventListener(btn, UIEventType.CLICK, Main_OnClick);
end

--事件监听
function Main_OnClick(GameObject)
